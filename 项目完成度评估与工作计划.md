class ResponseGuard:
    """响应验证器：检查 LLM 输出与工具结果的一致性"""

    @staticmethod
    def validate(response: str, tool_results: List[Dict]) -> bool:
        """验证数字一致性，容忍 ±5% 误差"""
        import re

        # 提取响应中的数字
        response_numbers = [float(n) for n in re.findall(r'\d+\.?\d*', response)]

        # 提取工具结果中的数字
        tool_numbers = []
        for result in tool_results:
            if result.get('success') and result.get('data'):
                data = result['data']
                if 'price' in data:
                    tool_numbers.append(data['price'])
                if 'change_pct' in data:
                    tool_numbers.append(abs(data['change_pct']))

        # 检查匹配
        for rn in response_numbers:
            matched = any(abs(rn - tn) / max(tn, 0.01) < 0.05 for tn in tool_numbers)
            if not matched and rn > 1:
                return False

        return True
```

class QueryRouter:
    """查询路由器：意图识别 + 模式选择"""

    def classify_intent(self, query: str) -> List[str]:
        """
        分类查询意图（可多个）：
        - MARKET_QUERY: 行情查询
        - KNOWLEDGE_QUERY: 知识问答
        - ANALYSIS_QUERY: 分析决策
        - NEWS_QUERY: 新闻事件
        """
        intents = []

        # 规则匹配
        if any(kw in query for kw in ["价格", "股价", "多少"]):
            intents.append("MARKET_QUERY")

        if any(kw in query for kw in ["能买吗", "建议", "分析"]):
            intents.append("ANALYSIS_QUERY")

        if any(kw in query for kw in ["什么是", "如何", "定义"]):
            intents.append("KNOWLEDGE_QUERY")

        if any(kw in query for kw in ["为什么", "原因", "新闻"]):
            intents.append("NEWS_QUERY")

        return intents or ["MARKET_QUERY"]

    def select_mode(self, query: str, intents: List[str]) -> str:
        """
        选择分析模式：
        - FAST: 简单查询（价格、基础信息）
        - DEEP: 复杂查询（分析、决策）
        """
        if "ANALYSIS_QUERY" in intents:
            return "DEEP"
        return "FAST"
```

class FastAnalyzer:
    """快速分析器：1-2秒响应"""

    def __init__(self):
        self.indicators = TechnicalIndicators()

    async def analyze(self, data: Dict) -> Dict:
        """
        快速分析流程：
        1. 计算技术指标（RSI, MACD）
        2. 判断趋势（上涨/下跌/震荡）
        3. 识别信号（超买/超卖/金叉/死叉）
        """
        result = {
            "mode": "FAST",
            "signals": {},
            "trend": {},
            "summary": ""
        }

        # 计算指标
        if "history" in data:
            prices = [p["close"] for p in data["history"]["data"]]
            result["signals"]["rsi"] = self.indicators.calculate_rsi(prices)
            result["signals"]["macd"] = self.indicators.calculate_macd(prices)

        # 判断趋势
        if "change" in data:
            change_pct = data["change"]["change_pct"]
            if change_pct > 5:
                result["trend"]["direction"] = "强势上涨"
            elif change_pct < -5:
                result["trend"]["direction"] = "强势下跌"
            else:
                result["trend"]["direction"] = "震荡整理"

        return result
```

            })

        # 估值风险
        if data.get("info", {}).get("pe_ratio", 0) > 50:
            risks.append({
                "type": "估值风险",
                "level": "高",
                "description": "市盈率偏高，估值存在泡沫"
            })

        return {
            "risks": risks,
            "overall_level": self._calculate_overall_level(risks)
        }
```

```python
class AgentCore:
    def __init__(self):
        # 原有组件
        self.model_manager = model_manager
        self.market_service = MarketDataService()
        self.rag_pipeline = HybridRAGPipeline()

        # 新增 Reasoning Layer
        self.query_router = QueryRouter()
        self.fast_analyzer = FastAnalyzer()
        self.deep_analyzer = DeepAnalyzer()
        self.decision_engine = DecisionEngine()
        self.risk_assessor = RiskAssessor()
        self.response_generator = ResponseGenerator()

    async def run(self, query: str, model_name: Optional[str] = None):
        """
        新流程：
        1. 意图识别 → 模式选择
        2. 并行收集数据
        3. 分析（Fast/Deep）
        4. 决策建议
        5. 风险评估
        6. 结构化响应
        """
        # 1. 意图识别
        intents = self.query_router.classify_intent(query)
        mode = self.query_router.select_mode(query, intents)

        # 2. 收集数据（并行）
        data = await self._collect_data_parallel(query, intents)

        # 3. 分析
        if mode == "FAST":
            analysis = await self.fast_analyzer.analyze(data)
        else:
            analysis = await self.deep_analyzer.analyze(data)

        # 4. 决策
        decision = self.decision_engine.generate_decision(analysis)

        # 5. 风险评估
        risks = self.risk_assessor.assess(analysis, data)

        # 6. 生成响应
        response = self.response_generator.generate(query, analysis, decision, risks)

        # 流式返回
        for section in response["sections"]:
            yield SSEEvent(type="section", data=section)
```
